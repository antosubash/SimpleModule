**Goal:** Let admins build multi-level public navigation menus from the Settings module, with page selection and home page designation.

**Architecture:** New `PublicMenuItemEntity` in Settings DB, `IPublicMenuProvider` abstraction in Core, `PageRegistryEmitter` in source generator. Settings implements the provider, Blazor layout consumes it. Home page is an internal URL rewrite middleware.

**Tech Stack:** EF Core (SQLite/PostgreSQL), ASP.NET Minimal APIs, React 19, Inertia.js, @dnd-kit for drag-and-drop, @simplemodule/ui components, Roslyn source generator (netstandard2.0), xUnit.v3 + FluentAssertions.

**Step 2: Create `PageRegistryEmitter`**

This emitter generates a static `PageRegistry` class that lists all discovered `IViewEndpoint` pages. It iterates `data.Modules`, and for each module with views, emits an `AvailablePage` entry with the page route (e.g., `"Products/Browse"`), the ViewPrefix (e.g., `"/products"`), and the module name.

The emitter must target netstandard2.0 patterns (no collection expressions, no raw string literals). Use `StringBuilder` like the existing emitters.

Reference `ViewPagesEmitter.cs` and `EndpointExtensionsEmitter.cs` for patterns. The `ModuleInfoRecord` has `ViewPrefix` (string) and `Views` (`ImmutableArray<ViewInfoRecord>`), where `ViewInfoRecord` has `FullyQualifiedName` and `Page` (the Inertia component name like `"Products/Browse"`).

```csharp
// framework/SimpleModule.Generator/Emitters/PageRegistryEmitter.cs
using System.Text;
using Microsoft.CodeAnalysis;
using Microsoft.CodeAnalysis.Text;

namespace SimpleModule.Generator;

internal sealed class PageRegistryEmitter : IEmitter
{
    public void Emit(SourceProductionContext context, DiscoveryData data)
    {
        var sb = new StringBuilder();
        sb.AppendLine("// <auto-generated/>");
        sb.AppendLine("using SimpleModule.Core.Menu;");
        sb.AppendLine("using System.Collections.Generic;");
        sb.AppendLine();
        sb.AppendLine("namespace SimpleModule.Core;");
        sb.AppendLine();
        sb.AppendLine("public static class PageRegistry");
        sb.AppendLine("{");
        sb.AppendLine(
            "    public static IReadOnlyList<AvailablePage> Pages { get; } = new AvailablePage[]"
        );
        sb.AppendLine("    {");

        foreach (var module in data.Modules)
        {
            if (module.Views.Length == 0)
                continue;

            var moduleName = module.ModuleName;

            for (var i = 0; i < module.Views.Length; i++)
            {
                var view = module.Views[i];
                sb.AppendLine(
                    string.Format(
                        "        new AvailablePage(\"{0}\", \"{1}\", \"{2}\"),",
                        view.Page,
                        module.ViewPrefix,
                        moduleName
                    )
                );
            }
        }

        sb.AppendLine("    };");
        sb.AppendLine("}");

        context.AddSource(
            "PageRegistry.g.cs",
            SourceText.From(sb.ToString(), Encoding.UTF8)
        );
    }
}
```

```csharp
// modules/Settings/src/Settings/Services/PublicMenuService.cs
using Microsoft.EntityFrameworkCore;
using Microsoft.Extensions.Caching.Memory;
using SimpleModule.Core.Menu;
using SimpleModule.Settings.Contracts;
using SimpleModule.Settings.Entities;

namespace SimpleModule.Settings.Services;

public class PublicMenuService(
    SettingsDbContext db,
    IMemoryCache cache
) : IPublicMenuProvider
{
    private const string MenuCacheKey = "PublicMenu_Tree";
    private const string HomeCacheKey = "PublicMenu_Home";
    private static readonly TimeSpan CacheDuration = TimeSpan.FromSeconds(60);

    public async Task<IReadOnlyList<PublicMenuItem>> GetMenuTreeAsync()
    {
        if (cache.TryGetValue(MenuCacheKey, out IReadOnlyList<PublicMenuItem>? cached) && cached is not null)
            return cached;

        var entities = await db.PublicMenuItems
            .Where(m => m.IsVisible)
            .OrderBy(m => m.SortOrder)
            .ToListAsync();

        var tree = BuildTree(entities, parentId: null);
        cache.Set(MenuCacheKey, tree, CacheDuration);
        return tree;
    }

    public async Task<string?> GetHomePageUrlAsync()
    {
        if (cache.TryGetValue(HomeCacheKey, out string? cached))
            return cached;

        var homePage = await db.PublicMenuItems
            .Where(m => m.IsHomePage && m.IsVisible)
            .Select(m => m.Url ?? m.PageRoute)
            .FirstOrDefaultAsync();

        cache.Set(HomeCacheKey, homePage, CacheDuration);
        return homePage;
    }

    public async Task<List<PublicMenuItemDto>> GetAllAsync()
    {
        var entities = await db.PublicMenuItems
            .OrderBy(m => m.SortOrder)
            .ToListAsync();

        return BuildDtoTree(entities, parentId: null);
    }

    public async Task<PublicMenuItemEntity> CreateAsync(CreateMenuItemRequest request)
    {
        if (request.ParentId.HasValue)
        {
            var depth = await GetDepthAsync(request.ParentId.Value);
            if (depth >= 3)
                throw new InvalidOperationException("Cannot exceed maximum depth of 3 levels.");
        }

        var maxOrder = await db.PublicMenuItems
            .Where(m => m.ParentId == request.ParentId)
            .MaxAsync(m => (int?)m.SortOrder) ?? -1;

        var entity = new PublicMenuItemEntity
        {
            ParentId = request.ParentId,
            Label = request.Label,
            Url = request.Url,
            PageRoute = request.PageRoute,
            Icon = request.Icon,
            CssClass = request.CssClass,
            OpenInNewTab = request.OpenInNewTab,
            IsVisible = request.IsVisible,
            IsHomePage = request.IsHomePage,
            SortOrder = maxOrder + 1,
            CreatedAt = DateTimeOffset.UtcNow,
            UpdatedAt = DateTimeOffset.UtcNow,
        };

        if (entity.IsHomePage)
            await ClearHomePageFlagAsync();

        db.PublicMenuItems.Add(entity);
        await db.SaveChangesAsync();
        InvalidateCache();
        return entity;
    }

    public async Task<PublicMenuItemEntity?> UpdateAsync(int id, UpdateMenuItemRequest request)
    {
        var entity = await db.PublicMenuItems.FindAsync(id);
        if (entity is null) return null;

        entity.Label = request.Label;
        entity.Url = request.Url;
        entity.PageRoute = request.PageRoute;
        entity.Icon = request.Icon;
        entity.CssClass = request.CssClass;
        entity.OpenInNewTab = request.OpenInNewTab;
        entity.IsVisible = request.IsVisible;
        entity.IsHomePage = request.IsHomePage;
        entity.UpdatedAt = DateTimeOffset.UtcNow;

        if (entity.IsHomePage)
            await ClearHomePageFlagAsync(entity.Id);

        await db.SaveChangesAsync();
        InvalidateCache();
        return entity;
    }

    public async Task<bool> DeleteAsync(int id)
    {
        var entity = await db.PublicMenuItems.FindAsync(id);
        if (entity is null) return false;

        db.PublicMenuItems.Remove(entity);
        await db.SaveChangesAsync();
        InvalidateCache();
        return true;
    }

    public async Task ReorderAsync(ReorderMenuItemsRequest request)
    {
        foreach (var item in request.Items)
        {
            var entity = await db.PublicMenuItems.FindAsync(item.Id);
            if (entity is null) continue;

            entity.ParentId = item.ParentId;
            entity.SortOrder = item.SortOrder;
            entity.UpdatedAt = DateTimeOffset.UtcNow;
        }

        await db.SaveChangesAsync();
        InvalidateCache();
    }

    public async Task SetHomePageAsync(int id)
    {
        await ClearHomePageFlagAsync();

        var entity = await db.PublicMenuItems.FindAsync(id);
        if (entity is null) return;

        entity.IsHomePage = true;
        entity.UpdatedAt = DateTimeOffset.UtcNow;
        await db.SaveChangesAsync();
        InvalidateCache();
    }

    public async Task ClearHomePageAsync()
    {
        await ClearHomePageFlagAsync();
        await db.SaveChangesAsync();
        InvalidateCache();
    }

    private async Task ClearHomePageFlagAsync(int? exceptId = null)
    {
        var homePages = await db.PublicMenuItems
            .Where(m => m.IsHomePage && (exceptId == null || m.Id != exceptId))
            .ToListAsync();

        foreach (var hp in homePages)
            hp.IsHomePage = false;
    }

    private async Task<int> GetDepthAsync(int parentId)
    {
        var depth = 1;
        var currentId = (int?)parentId;

        while (currentId.HasValue)
        {
            var parent = await db.PublicMenuItems
                .Where(m => m.Id == currentId)
                .Select(m => m.ParentId)
                .FirstOrDefaultAsync();

            if (parent is null) break;
            currentId = parent;
            depth++;
        }

        return depth;
    }

    private static List<PublicMenuItem> BuildTree(List<PublicMenuItemEntity> entities, int? parentId)
    {
        return entities
            .Where(e => e.ParentId == parentId)
            .Select(e => new PublicMenuItem
            {
                Label = e.Label,
                Url = e.Url ?? e.PageRoute ?? "#",
                Icon = e.Icon,
                CssClass = e.CssClass,
                OpenInNewTab = e.OpenInNewTab,
                IsHomePage = e.IsHomePage,
                Children = BuildTree(entities, e.Id),
            })
            .ToList();
    }

    private static List<PublicMenuItemDto> BuildDtoTree(List<PublicMenuItemEntity> entities, int? parentId)
    {
        return entities
            .Where(e => e.ParentId == parentId)
            .Select(e => new PublicMenuItemDto
            {
                Id = e.Id,
                ParentId = e.ParentId,
                Label = e.Label,
                Url = e.Url,
                PageRoute = e.PageRoute,
                Icon = e.Icon,
                CssClass = e.CssClass,
                OpenInNewTab = e.OpenInNewTab,
                IsVisible = e.IsVisible,
                IsHomePage = e.IsHomePage,
                SortOrder = e.SortOrder,
                Children = BuildDtoTree(entities, e.Id),
            })
            .ToList();
    }

    private void InvalidateCache()
    {
        cache.Remove(MenuCacheKey);
        cache.Remove(HomeCacheKey);
    }
}
```

**Step 2: Add middleware to `Program.cs`**

Insert after `app.UseAuthorization();` and before `app.MapRazorComponents`:

```csharp
// Home page rewrite: if a public menu item is designated as home page,
// rewrite GET / to that page's URL (internal rewrite, URL stays /)
app.Use(async (context, next) =>
{
    if (context.Request.Path == "/" && HttpMethods.IsGet(context.Request.Method))
    {
        var menuProvider = context.RequestServices.GetService<IPublicMenuProvider>();
        if (menuProvider is not null)
        {
            var homeUrl = await menuProvider.GetHomePageUrlAsync();
            if (homeUrl is not null && homeUrl != "/")
            {
                context.Request.Path = homeUrl;
            }
        }
    }

    await next();
});
```
